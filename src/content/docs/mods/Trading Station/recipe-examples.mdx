### One ingredient

	json={

## Machine Requirement

<CodeVariants
	json={
		`
  {
    "type": "trading_station:trading",
    "result": {
      "id": "minecraft:emerald_block",
      "count": 1
    },
    "ingredients": [
      {
        "ingredient": {
          "item": "minecraft:diamond"
        },
        "count": 5
      }
    ],
    "processingTime": 100,
    "requirements": [
      {
        "value": [
          "powered",
          "mechanical"
        ],
        "type": "trading_station:machine_id"
      }
    ]
  }
		`
	}
	kube={
		`
event.recipes.trading_station.trading(
    Item.of('minecraft:emerald_block'),
    [Item.of("5x minecraft:diamond")]
  )
  .processingTime(100)
  .requirements(MachineId.of(["powered","mechanical"]));

		`
	}

 />
