### Bat
<CodeVariants
	json={
		`
{
  "type": "create_mechanical_spawner:spawner",
  "input": {
    "type": "fluid_stack",
    "amount": 100,
    "fluid": "create_mechanical_spawner:spawn_fluid_bat"
  },
  "output": "minecraft:bat",
  "processingTime": 1000
}
		`
	}
	kube={
		`
		event.recipes.create_mechanical_spawner.spawner(
			"minecraft:bat", Fluid.of('create_mechanical_spawner:spawn_fluid_bat', 100)
		)
		.processingTime(1000);
		`
	}

  "customLoot": [
    {
      "id": "minecraft:nether_star"
    },
    {
      "count": 16,
      "id": "create:experience_nugget"
    },
    {
      "chance": 0.5,
      "count": 8,
      "id": "create:experience_nugget"
    }
  ],
  "input": {
    "type": "fluid_stack",
    "amount": 300,
    "fluid": "create_mechanical_spawner:spawn_fluid_wither"

  "processingTime": 5000
}
		`
	}
	kube={
		`
